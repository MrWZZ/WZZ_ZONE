元素内容


操作HTML

获取HTML内容
html()
获取匹配元素集合中第一个元素的HTML内容。

  var s = $("#id").html();


设置HTML内容
html(htmlString)
设置元素的HTML内容。

  $("#id").html("更换内容");


根据索引设置HTML内容
html(function(index,htmk))


操作文本

text()

  //获取文本内容
  var s = $("#id").text();

  //设置文本内容
  $("#id").text("更改");

  //根据索引设置文本内容
  $("#id").text(function(){});



操作值

val()

  //获取元素的值
  var v = $("#id").val();

  //设置值
  $("#id").val("value");

  //根据索引设置元素值
  $("#id").val(function(){});



元素属性

attr()

  //读取属性
  var v = $("#id").attr("key");

  //设置属性
  $("#id").attr("key",value);

  //根据索引设置属性
  $("#id").attr("key",function(){});


removeAttr()

  //删除属性
  $("#id").removeAttr("key");



元素样式

添加样式类

  addClass(class);
  addClass(function(){});


移除样式类

  removeClass(class);
  removeClass(function(){});


切换样式类

  //如果样式存在，则删除；如果样式不存在，则添加
  toggleClass(class);

  //switch：true->添加；false->删除
  toggleClass(class,switch);

  toggleClass(class,function(){});



元素CSS

css(cssname)

  //读取属性值
  var c = $("#id").css("color");

  //设置
  $("div").class("key","value");



元素CSS位置

offset()
可以控制元素相对于文档的当前坐标。

  //返回集合中第一个元素相对于文档的top和left坐标
  offset();

  //实例
  $("#panel").offset({top:50,left:400});
  //使用该方法时，如果他的position属性为static，则该属性将被更改为relative。


position()
用于获取相对于父元素的当前偏移量。
用法同上。

scrollLeft(),scrollTop()
分别用于获取元素的滚动条距离左端和顶端的距离。

  //获取值
  var l = $("#id").scrollLeft();
  var t = $("#id").scrollTop();

  //设置该元素内的滚动条滚动到指定位置
  $("textarea").scrollTop(50);
  $("textarea").scrollLeft(50);



元素CSS尺寸

height(),width()

  //获取高、宽
  var h = $("#id").height();
  var w =$("#id").width();

  //设置高、宽
  $("#id").height("300");
  $("#id").width("20%");
