        if(!(theRoot == null))

      //先序遍历,
      public void PreOrder(Node theRoot)
      {
        if(!(theRoot == null))
        {
          theRoot.DisplayNode();
          InOrder(theRoot.Left);

      public void PostOrder(Node theRoot)
      {
        if(!(theRoot == null))
        {
          InOrder(theRoot.Left);
          InOrder(theRoot.Right);
          theRoot.DisplayNode();
        }
      }
